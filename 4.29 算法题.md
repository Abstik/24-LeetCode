## [59. 螺旋矩阵 II](https://leetcode.cn/problems/spiral-matrix-ii/)

![image-20250429194006503](https://ting2.oss-cn-beijing.aliyuncs.com/picture/202504291940830.png)

```C
func generateMatrix(n int) [][]int {

	// 返回数组 并初始化
	ans := make([][]int, n)

	for i := range ans {
		ans[i] = make([]int, n)
	}

	up, down, left, right := 0, n-1, 0, n-1

	// 填入的数字 1 - n*n
	cur := 1

	for cur <= n*n {
        // //向右走的条件  行最上列不为最右列
		for i := left; i <= right; i++ {
			ans[up][i] = cur
			cur++
		}
		up++

        // 向下走的条件 列为最右 行不为最下
        for i := up; i <= down; i++ {
            ans[i][right] = cur
            cur++
        }
        right--

        // 向左走的条件 列不为最左  行为最下
        for i := right; i >= left; i-- {
            ans[down][i] = cur
            cur++
        }
        down--

        // 向上走的条件 列最左 行不为最上
        for i := down; i >= up; i-- {
            ans[i][left] = cur
            cur++
        }
        left++
	}
    return ans
}
```
